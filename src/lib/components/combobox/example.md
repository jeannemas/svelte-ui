### Usage

```html
<Combobox.Root>
  <Combobox.Input />

  <Combobox.Content>
    <Combobox.Group>
      <Combobox.GroupLabel />

      <Combobox.Item>
        <Combobox.ItemIndicator />

        <Combobox.Label />
      </Combobox.Item>
    </Combobox.Group>

    <Combobox.Separator />

    <Combobox.Item>
      <Combobox.ItemIndicator />

      <Combobox.Label />
    </Combobox.Item>
  </Combobox.Content>

  <Combobox.HiddenInput />
</Combobox.Root>
```
